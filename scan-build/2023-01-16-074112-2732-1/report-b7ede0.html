396 FAIL ("bech32 encoded address of incorrect length"){ ;; ; os_longjmp(0x1); };
397 }
398 append_ctx_msg(ctx, buf2, BECH32_ADDRSTR_LEN(3 + 1 + 32 + 6));
399 } else {
400 assert(readlen == ZIL_AMOUNT_GASPRICE_BYTES)if (readlen == 16) {} else { ;; ; os_longjmp(0x1); };
401 // It is either gasprice or amount. a uint128_t value.
402 // Convert to decimal, appending a '\0'.
403 // ZIL data is big-endian, we need little-endian here.
404 for (int i = 0; i < 4; i++) {
405 // The upper 64b and lower 64b themselves aren't swapped, just within them.
406 // Upper uint64 is converted to little endian.
407 uint8_t t = buf[i];
408 buf[i] = buf[7-i];
409 buf[7-i] = t;
410 // lower uint64 is converted to little endian.
411 t = buf[8+i];
412 buf[8+i] = buf[15-i];
413 buf[15-i] = t;
414 }
415 CHECK_CANARY;
416 // UINT128 can have a maximum of 39 decimal digits. When we convert
417 // "Qa" values to "Zil", we may have to append "0." at the start.
418 // So a total of 39 + 2 + '\0' = 42.
419 if (tostring128((uint128_t*)buf, 10, buf2, buf2size)) {
420 PRINTF("128b to decimal converted value: %s\n", buf2);
421 CHECK_CANARY;
422 qa_to_zil(buf2, buf, bufsize);
423 PRINTF("Qa converted to Zil: %s\n", buf);
424 append_ctx_msg(ctx, buf, strlen(buf));
425 CHECK_CANARY;
426 } else {
427 FAIL("Error converting 128b unsigned to decimal"){ ;; ; os_longjmp(0x1); };
428 }
429 }
430 CHECK_CANARY;
431 append_ctx_msg(ctx, " ", 1);
432 PRINTF("pb_read: read %d bytes\n", readlen);
433 } else {
434 PRINTF("pb_read failed\n");
435 return false0;
436 }
437
438 CHECK_CANARY;
439 return true1;
440}
441// Sign the txn, also deserializes parts of it. May call io_exchange multiple times.
442// Output: 1. Display message will be populated in ctx->msg.
443// 2. Signature will be populated in ctx->signature.
444static bool_Bool sign_deserialize_stream(const uint8_t *txn1, int txn1Len, int hostBytesLeft)
445{
446 // Initialize stream data.
447 memcpy(ctx->sd.buf, txn1, txn1Len);
448 ctx->sd.nextIdx = 0; ctx->sd.len = txn1Len; ctx->sd.hostBytesLeft = hostBytesLeft;
449 assert(hostBytesLeft <= ZIL_MAX_TXN_SIZE - txn1Len)if (hostBytesLeft <= 8388608 - txn1Len) {} else { ;; ; os_longjmp
(0x1); };
450 // Setup the stream.
451 pb_istream_t stream = { istream_callback, &ctx->sd, hostBytesLeft + txn1Len, NULL((void*)0) };
452
453 // Initialize the display message.
454 ctx->msgLen = 0;
455
456 CHECK_CANARY;
457 // Initialize schnorr signing, continue with what we have so far.
458 deriveAndSignInit(&ctx->ecs, ctx->keyIndex);
459 CHECK_CANARY;
460 deriveAndSignContinue(&ctx->ecs, txn1, txn1Len);
461 CHECK_CANARY;
462
463 // Initialize protobuf Txn structs.
464 memset(&ctx->txn, 0, sizeof(ctx->txn));
465 // Set callbacks for handling the fields that what we need.
466 ctx->txn.toaddr.funcs.decode = decode_callback;
467 // Since we're using the same callback for amount and gasprice,
468 // but the tag for both will be set to "ByteArray", we differentiate with "arg".
469 ctx->txn.amount.data.funcs.decode = decode_callback;
470 ctx->txn.amount.data.arg = (void*)ProtoTransactionCoreInfo_amount_tag5;
471 ctx->txn.gasprice.data.funcs.decode = decode_callback;
472 ctx->txn.gasprice.data.arg = (void*)ProtoTransactionCoreInfo_gasprice_tag6;
473 // Set a decoder for the data field of our transaction.
474 ctx->txn.data.funcs.decode = decode_txn_data;
475
476 CHECK_CANARY;
477
478 // Start decoding (and signing).
479 if (pb_decode(&stream, ProtoTransactionCoreInfo_fields, &ctx->txn)) {
480 PRINTF ("pb_decode successful\n");
481 deriveAndSignFinish(&ctx->ecs, ctx->keyIndex, ctx->signature, SCHNORR_SIG_LEN_RS64);
482 PRINTF ("sign_deserialize_stream: signature: 0x%.*h\n", SCHNORR_SIG_LEN_RS, ctx->signature);
483 } else {
484 PRINTF ("pb_decode failed\n");
485 return false0;
486 }
487
488 CHECK_CANARY;
489 return true1;
490}
